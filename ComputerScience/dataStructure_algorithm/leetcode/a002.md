树的遍历  
### 深度优先  
```
private void depthFirstVisit(MultiTreeNode<D> node) {
    if (node == null) {
        return;
    }
    Stack<MultiTreeNode<D>> stack = new Stack<>();
    stack.push(node);
    while (!stack.isEmpty()) {
        MultiTreeNode<D> node1 = stack.pop();
        if (node1 == null) {
            continue;
        }
        LogTrack.w(node1.data);
        List<MultiTreeNode<D>> list = node1.childList;
        int size = list == null ? 0 : list.size();
        for (int i = 0; i < size; i++) {
            stack.push(list.get(i));
        }
    }

}

### 广度优先  
```
private void breadthFirstVisit(MultiTreeNode<D> node) {
    if (node == null) {
        return;
    }
    Queue<MultiTreeNode<D>> queue = new LinkedList<>();
    queue.add(node);
    while (!queue.isEmpty()) {
        MultiTreeNode<D> node1 = queue.poll();
        if (node1 == null) {
            continue;
        }
        LogTrack.w(node1.data);
        List<MultiTreeNode<D>> list = node1.childList;
        int size = list == null ? 0 : list.size();
        for (int i = 0; i < size; i++) {
            queue.add(list.get(i));
        }
    }
}
