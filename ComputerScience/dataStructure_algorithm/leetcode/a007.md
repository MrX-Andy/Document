public static int test(int array[], int target) {
        if (array == null || array.length <= 0) {
            return Integer.MIN_VALUE;
        }
        int start = 0;
        int end = array.length - 1;
        while (start <= end) {
            int mid = (start + end) / 2;
            int item = array[mid];
            if (item > target) {
                end = mid - 1;
            } else if (item < target) {
                start = mid + 1;
            } else {
                return array[mid];
            }
        }
        if (start < array.length) {
            return array[start];
        }
        return Integer.MIN_VALUE;
    }
