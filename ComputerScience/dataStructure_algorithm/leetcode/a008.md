### 单链表反转  
```
/**
 * 单链表反转
 */
private Node reverseList(Node node) {
    if (node == null) {
        return null;
    }
    Node pre = null;
    Node curr = node;
    while (curr != null) {
        Node next = curr.next;
        curr.next = pre;
        pre = curr;
        curr = next;
    }
    return pre;
}

### 双向链表插入

public void addFirst(String e) {
    Node[] nodes = addFirst(first, last, e);
    first = nodes[0];
    last = nodes[1];
}

private Node[] addFirst(Node first, Node last, String e) {
    final Node f = first;
    final Node node = new Node(e, null, f);
    first = node;
    if (f == null) {
        last = node;
    } else {
        f.pre = node;
    }
    size++;
    modCount++;
    return new Node[]{first, last};
}

public void addLast(String e) {
    Node[] nodes = addLast(first, last, e);
    first = nodes[0];
    last = nodes[1];
}

private Node[] addLast(Node first, Node last, String e) {
    final Node l = last;
    final Node node = new Node(e, l, null);
    last = node;
    if (l == null) {
        first = node;
    } else {
        l.next = node;
    }
    size++;
    modCount++;
    return new Node[]{first, last};
}

### 双向链表反转  
```
/**
 * 指针反转
 */
private Node[] reverseList(Node first, Node last) {
    if (first == null || last == null) {
        return new Node[]{first, last};
    }
    Node temp = null;
    Node curr = first;
    last = first;
    while (curr != null) {
        temp = curr.pre;
        curr.pre = curr.next;
        curr.next = temp;
        curr = curr.pre;
    }
    if (temp != null) {
        first = temp.pre;
    }
    return new Node[]{first, last};
}

/**
 * 数据反转
 */
public void reverseData(Node first, Node last) {
    if (first == null || last == null) {
        return;
    }
    Node f = first;
    Node l = last;
    while (f != l) {
        String data = f.data;
        f.data = l.data;
        l.data = data;
        f = f.next;
        if (f == l) {
            break;
        }
        l = l.pre;
    }
}
