单链表翻转  
```
/**
 * 单链表反转
 */
private Node reverseList(Node node) {
    if (node == null) {
        return node;
    }
    Node pre = null;
    Node curr = node;
    while (curr != null) {
        Node next = curr.next;
        curr.next = pre;
        pre = curr;
        curr = next;
    }
    return pre;
}
```
