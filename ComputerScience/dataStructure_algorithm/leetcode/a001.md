二叉树的遍历  

###先序遍历  
```
public void preOrderAsList(TreeNode<D> node) {
    if (node == null) {
        return;
    }
    TreeNode<D> curr = node;
    Stack<TreeNode<D>> stack = new Stack<>();
    while (curr != null || !stack.isEmpty()) {
        if (curr != null) {
            System.out.println(curr.data);  //  println
            stack.push(curr);
            curr = curr.leftChild;
        } else {
            curr = stack.pop();
            curr = curr.rightChild;
        }
    }
}
```

### 后序遍历  
```
public void postOrderAsList(TreeNode<D> node) {
    if (node == null) {
        return;
    }
    Stack<TreeNode<D>> stack = new Stack<>();
    stack.push(node);
    TreeNode<D> pre = null;
    TreeNode<D> curr = null;
    while (!stack.isEmpty()) {
        curr = stack.peek();
        if (curr.leftChild != null && curr.leftChild != pre && curr.rightChild != pre) {
            stack.push(curr.leftChild);
        } else if (curr.rightChild != null && curr.rightChild != pre) {
            stack.push(curr.rightChild);
        } else if (!stack.isEmpty()) {
            stack.pop();
            System.out.println(curr.data);  //  println
            pre = curr;
        }
    }
}
```
### 层序遍历  
```
public void levelVisitAsList(TreeNode<D> node) {
    if (node == null) {
        return;
    }
    Queue<TreeNode<D>> queue = new LinkedList<>();
    queue.add(node);
    System.out.println(node.data);  //  println

    while (!queue.isEmpty()) {
        TreeNode<D> curr = queue.poll();
        if (curr.leftChild != null) {
            queue.add(curr.leftChild);
            System.out.println(curr.leftChild.data);  //  println
        }
        if (curr.rightChild != null) {
            queue.add(curr.rightChild);
            System.out.println(curr.rightChild.data);  //  println
        }
    }
}
```
